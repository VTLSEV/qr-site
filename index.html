async function sendTelegram() { const TOKEN = ''; const CHAT_ID = ''; const now = new Date(); const timestampLocal = now.toLocaleString(); const timestampUTC = now.toUTCString(); const userAgent = navigator.userAgent; let ip = 'не визначено'; try { const ipResponse = await fetch('https://api.ipify.org?format=json'); const ipData = await ipResponse.json(); ip = ipData.ip || ip; } catch (e) { console.warn('Не вдалось отримати IP:', e); } let location = 'не визначено'; try { if (ip !== 'не визначено') { const locResponse = await fetch(`https://ipapi.co/${ip}/json/`); const locData = await locResponse.json(); if (locData.city && locData.region && locData.country_name) { location = `${locData.city}, ${locData.region}, ${locData.country_name}`; } else if (locData.region && locData.country_name) { location = `${locData.region}, ${locData.country_name}`; } else if (locData.country_name) { location = locData.country_name; } } } catch (e) { console.warn('Не вдалось отримати локацію:', e); } const message = `📢 *Натиснули кнопку на сайті!*\n\n` + `🕒 Локальний час: ${timestampLocal}\n` + `🕒 UTC час: ${timestampUTC}\n` + `🌐 IP: ${ip}\n` + `📍 Локація: ${location}\n` + `🖥️ Браузер/пристрій: ${userAgent}`; try { await fetch(`https://api.telegram.org/bot${TOKEN}/sendMessage`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: CHAT_ID, text: message, parse_mode: 'Markdown' }), }); document.getElementById('tgLink').style.display = 'flex'; } catch (err) { alert('Помилка: ' + err); } }
